```powershell
SOLOv2-TensorRT$ cd build
```

```powershell
 ./convertModel ./solov2_r50_fpn_8gpu_3x.onnx -g ./seg_coco_permute.bin
```

```powershell
SOLOv2-TensorRT/build$ make clean
```
